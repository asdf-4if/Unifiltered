        // Tab functionality

        function showTab(tabName) {

            document.querySelectorAll('.tab-content').forEach(tab => {

                tab.classList.remove('active');

            });

            

            document.querySelectorAll('.nav-tab').forEach(tab => {

                tab.classList.remove('active');

            });

            

            document.getElementById(tabName).classList.add('active');

            // Find the button that was clicked and add the active class to it

            const clickedButton = Array.from(document.querySelectorAll('.nav-tab')).find(btn => btn.onclick.toString().includes(`'${tabName}'`));

            if (clickedButton) {

                clickedButton.classList.add('active');

            }

            

            if (tabName === 'magazine') {

                loadMagazineIssues();

            } else if (tabName === 'submissions') {

                loadSubmissions();

            }

        }

        function setupFileUpload() {

                fileUpload.classList.add('dragover');

            fileUpload.addEventListener('dragleave', () => {

                fileUpload.classList.remove('dragover');

            });

                e.preventDefault();

                handleFiles(e.dataTransfer.files);

            });

            

            fileInput.addEventListener('change', (e) => {

                handleFiles(e.target.files);

            });

        }

        function handleFiles(files) {

            for (let file of files) {

                if (file.size > 10 * 1024 * 1024) {

                    // Use a custom message box instead of alert

                    displayMessageBox(`File "${file.name}" is too large. Maximum size is 10MB.`);

                    continue;

                }

                selectedFiles.push(file);

            }

            updateFileList();

        }


        function updateFileList() {

            const fileList = document.getElementById('fileList');

            fileList.innerHTML = '';

            

            selectedFiles.forEach((file, index) => {

                const fileItem = document.createElement('div');

                fileItem.className = 'file-item';

                fileItem.innerHTML = `

                    <span class="file-item-icon">📄</span>

                    <span class="file-item-name">${file.name}</span>

                    <span class="file-item-size">${(file.size / 1024).toFixed(1)} KB</span>

                    <button onclick="removeFile(${index})" style="margin-left: 10px; background: #dc3545; color: white; border: none; border-radius: 4px; padding: 4px 8px; cursor: pointer;">Remove</button>

                `;

                fileList.appendChild(fileItem);

            });

        }


        function removeFile(index) {

            selectedFiles.splice(index, 1);

            updateFileList();

        }

        function deleteMagazine(magazineId) {

            // Use a custom confirmation dialog instead of confirm

            displayConfirmation('Are you sure you want to delete this magazine issue?', () => {

                if (!isAdmin) return;

                

                const storedMagazines = JSON.parse(localStorage.getItem('magazines') || '[]');

                const updatedMagazines = storedMagazines.filter(m => m.id !== magazineId);

            });

        }


        // Submissions functionality

        function loadSubmissions() {

            const submissionList = document.getElementById('submissionList');

            const storedSubmissions = JSON.parse(localStorage.getItem('submissions') || '[]');

            let userSubmissions = storedSubmissions;

            

            if (!isAdmin) {

                userSubmissions = storedSubmissions.filter(s => s.userEmail === currentUser.email);

            }

            

            if (userSubmissions.length === 0) {

                submissionList.innerHTML = `<p style="text-align: center; color: #666; padding: 40px;">

                    No submissions yet. <a href="#" onclick="showTab('submit')" style="color: #667eea;">Submit your work</a>

                </p>`;

                return;

            }

            

            submissionList.innerHTML = '';

            userSubmissions.forEach(submission => {

                const submissionDiv = document.createElement('div');

                submissionDiv.className = 'submission-item';

                submissionDiv.innerHTML = `

                    <div class="submission-meta">

                        <span class="submission-type">${submission.workType}</span>

                        <span class="status-badge status-${submission.status.toLowerCase().replace(' ', '-')}">${submission.status}</span>

                        <span>${new Date(submission.submittedAt).toLocaleDateString()}</span>

                    </div>

                    <h3>${submission.workTitle}</h3>

                    ${isAdmin ? `<p><strong>Submitted by:</strong> ${submission.userEmail}</p>` : ''}

                    <p>${submission.workDescription}</p>

                    ${submission.websiteLink ? `<p><strong>Website:</strong> <a href="${submission.websiteLink}" target="_blank">${submission.websiteLink}</a></p>` : ''}

                    ${submission.files.length > 0 ? `<p><strong>Files:</strong> ${submission.files.map(f => f.name).join(', ')}</p>` : ''}

                    ${isAdmin ? `<div class="admin-actions">

                        <button class="btn-success" onclick="updateSubmissionStatus('${submission.id}', 'Approved')">Approve</button>

                        <button class="btn-danger" onclick="updateSubmissionStatus('${submission.id}', 'Rejected')">Reject</button>

                        <button onclick="deleteSubmission('${submission.id}')">Delete</button>

                    </div>` : ''}

                `;

                submissionList.appendChild(submissionDiv);

            });

        }


        function updateSubmissionStatus(submissionId, status) {

            if (!isAdmin) return;

            

            const storedSubmissions = JSON.parse(localStorage.getItem('submissions') || '[]');

            const submissionIndex = storedSubmissions.findIndex(s => s.id === submissionId);

            if (submissionIndex !== -1) {

                storedSubmissions[submissionIndex].status = status;

                localStorage.setItem('submissions', JSON.stringify(storedSubmissions));

                loadSubmissions();

            }

        }


        function deleteSubmission(submissionId) {

            // Use a custom confirmation dialog instead of confirm

            displayConfirmation('Are you sure you want to delete this submission?', () => {

                if (!isAdmin) return;

                

                const storedSubmissions = JSON.parse(localStorage.getItem('submissions') || '[]');

                const updatedSubmissions = storedSubmissions.filter(s => s.id !== submissionId);

                localStorage.setItem('submissions', JSON.stringify(updatedSubmissions));

                loadSubmissions();

            });

        }


        // Custom Message Box and Confirmation Dialog

        function displayMessageBox(message) {

            const modal = document.createElement('div');

            modal.className = 'modal';

            modal.innerHTML = `

                <div class="modal-content" style="max-width: 400px; text-align: center;">

                    <span class="close" onclick="this.closest('.modal').remove()">&times;</span>

                    <p>${message}</p>

                    <button class="btn" onclick="this.closest('.modal').remove()" style="margin-top: 20px;">OK</button>

                </div>

            `;

            document.body.appendChild(modal);

            modal.style.display = 'block';

        }


        function displayConfirmation(message, onConfirm) {

            const modal = document.createElement('div');

            modal.className = 'modal';

            modal.innerHTML = `

                <div class="modal-content" style="max-width: 400px; text-align: center;">

                    <span class="close" onclick="this.closest('.modal').remove()">&times;</span>

                    <p>${message}</p>

                    <div style="margin-top: 20px; display: flex; justify-content: center; gap: 10px;">

                        <button class="btn btn-danger" onclick="this.closest('.modal').remove();">Cancel</button>

                        <button class="btn btn-success" onclick="this.closest('.modal').remove(); ${onConfirm.name} ? ${onConfirm.name}() : onConfirm();">Confirm</button>

                    </div>

                </div>

            `;

            document.body.appendChild(modal);

            modal.style.display = 'block';

        }

            }
